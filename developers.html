javascript • craftly-sdk-example.js 
import { CraftlyRobot } from '@craftly/sdk';

const robot = new CraftlyRobot({
  apiKey: 'YOUR_API_KEY',
  region: 'bd-dhaka-1'
});

async function buildApp() {
  const app = await robot.createApp({
    name: 'StudyBuddy',
    template: 'education-assistant',
    language: 'bn'
  });
  
  console.log(`App created: ${app.url}`);
}

buildApp();
